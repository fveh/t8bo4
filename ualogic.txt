function bo4_SetMaxStats()
{
    self stats::set_stat_global(#"rankxp", level.rankxpcap, 1); // Max XP
    self stats::set_stat_global(#"rank", level.maxrank, 1);     // Max Rank
    self stats::set_stat_global(#"plevel", level.maxprestige, 1); // Max Prestige

    // Optional Paragon
    self stats::set_stat_global(#"paragon_rank", 1000, 1);
    self stats::set_stat_global(#"paragon_icon_id", 1, 1);

    // Sofort Anwenden
    self.pers[#"rankxp"] = level.rankxpcap;
    self.pers[#"rank"]   = level.maxrank;
    self.pers[#"plevel"] = level.maxprestige;
    self setrank(level.maxrank, level.maxprestige);
    self iprintln("ğŸ–ï¸ Max Stats gesetzt!");
}
